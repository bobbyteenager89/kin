# Kin MVP Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a personal friend CRM with timeline dashboard, Google Calendar sync, Gmail email parsing, and a public calendar library.

**Architecture:** Next.js 15 App Router with server components, Clerk auth (Google OAuth), Neon Postgres via Drizzle ORM, Claude API for email parsing and AI calendar builder. Design ported from existing Vite prototype in this repo.

**Tech Stack:** Next.js 15, Clerk, Drizzle ORM, Neon Postgres, Google Calendar API, Gmail API, Claude API, CSS Modules, chrono-node, web-push

- [ ] **Step 6: Add user sync webhook**

When a user signs up via Clerk, we need to create a row in our `users` table.

Create `app/api/webhooks/clerk/route.ts`:
```ts
import { Webhook } from 'svix';
import { headers } from 'next/headers';
import { WebhookEvent } from '@clerk/nextjs/server';
import { db } from '@/lib/db';
import { users } from '@/lib/db/schema';

export async function POST(req: Request) {
  const WEBHOOK_SECRET = process.env.CLERK_WEBHOOK_SECRET;
  if (!WEBHOOK_SECRET) throw new Error('Missing CLERK_WEBHOOK_SECRET');

  const headerPayload = await headers();
  const svix_id = headerPayload.get('svix-id');
  const svix_timestamp = headerPayload.get('svix-timestamp');
  const svix_signature = headerPayload.get('svix-signature');

  if (!svix_id || !svix_timestamp || !svix_signature) {
    return new Response('Missing svix headers', { status: 400 });
  }

  const payload = await req.json();
  const body = JSON.stringify(payload);

  const wh = new Webhook(WEBHOOK_SECRET);
  let evt: WebhookEvent;

  try {
    evt = wh.verify(body, {
      'svix-id': svix_id,
      'svix-timestamp': svix_timestamp,
      'svix-signature': svix_signature,
    }) as WebhookEvent;
  } catch {
    return new Response('Invalid signature', { status: 400 });
  }

  if (evt.type === 'user.created') {
    await db.insert(users).values({ id: evt.data.id }).onConflictDoNothing();
  }

  return new Response('OK', { status: 200 });
}
```

- [ ] **Step 3: Create server actions**

`lib/actions/persons.ts`:
```ts
'use server';

import { auth } from '@clerk/nextjs/server';
import { db } from '@/lib/db';
import { persons, children, events } from '@/lib/db/schema';
import { personSchema } from '@/lib/validators/person';
import { eq, and } from 'drizzle-orm';
import { revalidatePath } from 'next/cache';

export async function createPerson(input: unknown) {
  const { userId } = await auth();
  if (!userId) throw new Error('Unauthorized');

  const data = personSchema.parse(input);
  const { children: childrenData, ...personData } = data;

  const [person] = await db.insert(persons).values({
    ...personData,
    ownerUserId: userId,
    tags: data.tags ?? [],
  }).returning();

  if (childrenData?.length) {
    await db.insert(children).values(
      childrenData.map((c) => ({ ...c, personId: person.id }))
    );
  }

  // Auto-create birthday event if birthday provided
  if (data.birthday) {
    await db.insert(events).values({
      userId,
      personId: person.id,
      type: 'birthday',
      title: `${data.name}'s Birthday`,
      date: data.birthday,
      recurring: true,
      source: 'manual',
    });
  }

  // Auto-create anniversary event if provided
  if (data.weddingAnniversary) {
    await db.insert(events).values({
      userId,
      personId: person.id,
      type: 'anniversary',
      title: `${data.name}'s Anniversary`,
      date: data.weddingAnniversary,
      recurring: true,
      source: 'manual',
    });
  }

  revalidatePath('/');
  return person;
}

export async function updatePerson(personId: string, input: unknown) {
  const { userId } = await auth();
  if (!userId) throw new Error('Unauthorized');

  const data = personSchema.parse(input);
  const { children: childrenData, ...personData } = data;

  await db.update(persons)
    .set({ ...personData, tags: data.tags ?? [], updatedAt: new Date() })
    .where(and(eq(persons.id, personId), eq(persons.ownerUserId, userId)));

  // Replace children
  if (childrenData) {
    await db.delete(children).where(eq(children.personId, personId));
    if (childrenData.length) {
      await db.insert(children).values(
        childrenData.map((c) => ({ ...c, personId }))
      );
    }
  }

  // Sync birthday/anniversary events: delete old ones, create new ones
  await db.delete(events).where(
    and(eq(events.personId, personId), eq(events.userId, userId), eq(events.source, 'manual'))
  );

  if (data.birthday) {
    await db.insert(events).values({
      userId, personId, type: 'birthday',
      title: `${data.name}'s Birthday`, date: data.birthday,
      recurring: true, source: 'manual',
    });
  }
  if (data.weddingAnniversary) {
    await db.insert(events).values({
      userId, personId, type: 'anniversary',
      title: `${data.name}'s Anniversary`, date: data.weddingAnniversary,
      recurring: true, source: 'manual',
    });
  }

  revalidatePath('/');
  revalidatePath(`/friends/${personId}`);
}

export async function deletePerson(personId: string) {
  const { userId } = await auth();
  if (!userId) throw new Error('Unauthorized');

  await db.delete(persons)
    .where(and(eq(persons.id, personId), eq(persons.ownerUserId, userId)));

  revalidatePath('/');
}

export async function markCaughtUp(personId: string) {
  const { userId } = await auth();
  if (!userId) throw new Error('Unauthorized');

  await db.update(persons)
    .set({ lastCaughtUp: new Date(), updatedAt: new Date() })
    .where(and(eq(persons.id, personId), eq(persons.ownerUserId, userId)));

  revalidatePath(`/friends/${personId}`);
}
```

import { db } from '@/lib/db';
import { events, persons } from '@/lib/db/schema';
import { eq, and, gte, asc } from 'drizzle-orm';

export async function getUpcomingEvents(userId: string, limit = 10) {
  const today = new Date().toISOString().split('T')[0];

  // For recurring events, we need to compute the next occurrence
  // For MVP, fetch all events for this user and compute in JS
  const allEvents = await db.query.events.findMany({
    where: eq(events.userId, userId),
  });

  const upcoming = allEvents
    .map((event) => {
      if (event.recurring) {
        // Compute next occurrence this year or next
        const eventDate = new Date(event.date);
        const now = new Date();
        const thisYear = new Date(now.getFullYear(), eventDate.getMonth(), eventDate.getDate());
        const nextYear = new Date(now.getFullYear() + 1, eventDate.getMonth(), eventDate.getDate());
        const nextOccurrence = thisYear >= now ? thisYear : nextYear;
        return { ...event, nextDate: nextOccurrence };
      }
      return { ...event, nextDate: new Date(event.date) };
    })
    .filter((e) => e.nextDate >= new Date())
    .sort((a, b) => a.nextDate.getTime() - b.nextDate.getTime())
    .slice(0, limit);

  return upcoming;
}

export async function getEventsForPerson(personId: string, userId: string) {
  return db.query.events.findMany({
    where: and(eq(events.personId, personId), eq(events.userId, userId)),
    orderBy: [asc(events.date)],
  });
}

'use client';

import { useState } from 'react';
import * as chrono from 'chrono-node';

interface DateInputProps {
  label: string;
  value: string;
  onChange: (isoDate: string) => void;
}

export function DateInput({ label, value, onChange }: DateInputProps) {
  const [text, setText] = useState(value);

  function handleBlur() {
    if (!text) { onChange(''); return; }
    const parsed = chrono.parseDate(text);
    if (parsed) {
      // Use sentinel year 1904 if no year was explicitly mentioned
      const hasYear = /\d{4}/.test(text);
      const date = hasYear
        ? parsed
        : new Date(1904, parsed.getMonth(), parsed.getDate());
      onChange(date.toISOString().split('T')[0]);
    }
  }

  return (
    <div>
      <label className="detail-label">{label}</label>
      <input
        type="text"
        value={text}
        onChange={(e) => setText(e.target.value)}
        onBlur={handleBlur}
        placeholder="e.g. March 15 or 1992-03-15"
        className="mock-input"
      />
    </div>
  );
}

import { auth, clerkClient } from '@clerk/nextjs/server';

async function getGoogleToken(userId: string): Promise<string | null> {
  const client = await clerkClient();
  const tokens = await client.users.getUserOauthAccessToken(userId, 'oauth_google');
  return tokens.data[0]?.token ?? null;
}

export async function pushEventToGoogleCalendar(
  userId: string,
  event: { title: string; date: string; recurring: boolean; description?: string }
) {
  const token = await getGoogleToken(userId);
  if (!token) return null;

  // Create or find "Kin — Birthdays" calendar
  // Use separate calendars per event type
  const calendarName = event.recurring
    ? (event.title.includes('Anniversary') ? 'Kin — Anniversaries' : 'Kin — Birthdays')
    : 'Kin — Events';
  const calendarId = await getOrCreateKinCalendar(token, calendarName);

  const body: Record<string, unknown> = {
    summary: event.title,
    description: event.description,
    start: { date: event.date },
    end: { date: event.date },
  };

  if (event.recurring) {
    body.recurrence = ['RRULE:FREQ=YEARLY'];
  }

  const res = await fetch(
    `https://www.googleapis.com/calendar/v3/calendars/${encodeURIComponent(calendarId)}/events`,
    {
      method: 'POST',
      headers: { Authorization: `Bearer ${token}`, 'Content-Type': 'application/json' },
      body: JSON.stringify(body),
    }
  );

  if (!res.ok) {
    console.error('Google Calendar push failed:', await res.text());
    return null;
  }

  return (await res.json()).id;
}

async function getOrCreateKinCalendar(token: string, name: string): Promise<string> {
  // List calendars, find existing Kin calendar
  const listRes = await fetch(
    'https://www.googleapis.com/calendar/v3/users/me/calendarList',
    { headers: { Authorization: `Bearer ${token}` } }
  );
  const list = await listRes.json();
  const existing = list.items?.find((c: { summary: string }) => c.summary === name);
  if (existing) return existing.id;

  // Create new calendar
  const createRes = await fetch(
    'https://www.googleapis.com/calendar/v3/calendars',
    {
      method: 'POST',
      headers: { Authorization: `Bearer ${token}`, 'Content-Type': 'application/json' },
      body: JSON.stringify({ summary: name }),
    }
  );
  return (await createRes.json()).id;
}

import { db } from '@/lib/db';
import { events, users } from '@/lib/db/schema';
import { eq, isNull } from 'drizzle-orm';
import { pushEventToGoogleCalendar } from '@/lib/google/calendar';

export async function GET(req: Request) {
  // Verify cron secret
  const authHeader = req.headers.get('authorization');
  if (authHeader !== `Bearer ${process.env.CRON_SECRET}`) {
    return new Response('Unauthorized', { status: 401 });
  }

  // Find events without a Google Calendar ID
  const unsyncedEvents = await db.query.events.findMany({
    where: isNull(events.googleCalendarEventId),
  });

  for (const event of unsyncedEvents) {
    const gcalId = await pushEventToGoogleCalendar(event.userId, {
      title: event.title,
      date: event.date,
      recurring: event.recurring,
      description: event.description ?? undefined,
    });

    if (gcalId) {
      await db.update(events)
        .set({ googleCalendarEventId: gcalId })
        .where(eq(events.id, event.id));
    }
  }

  return Response.json({ synced: unsyncedEvents.length });
}

'use server';

import { auth } from '@clerk/nextjs/server';
import { db } from '@/lib/db';
import { importSuggestions, persons, events } from '@/lib/db/schema';
import { eq, and } from 'drizzle-orm';
import { revalidatePath } from 'next/cache';

export async function approveImport(importId: string) {
  const { userId } = await auth();
  if (!userId) throw new Error('Unauthorized');

  const suggestion = await db.query.importSuggestions.findFirst({
    where: and(eq(importSuggestions.id, importId), eq(importSuggestions.userId, userId)),
  });
  if (!suggestion || suggestion.status !== 'pending') return;

  const data = suggestion.extractedData as {
    hostName?: string;
    eventName?: string;
    date?: string;
    location?: string;
  };

  // Fuzzy match against existing persons
  let personId: string | undefined;
  if (data.hostName) {
    const existing = await db.query.persons.findFirst({
      where: and(
        eq(persons.ownerUserId, userId),
        eq(persons.name, data.hostName), // Exact match for MVP; fuzzy in v2
      ),
    });
    personId = existing?.id;
  }

  // Create person if no match
  if (!personId && data.hostName) {
    const [newPerson] = await db.insert(persons).values({
      ownerUserId: userId,
      name: data.hostName,
    }).returning();
    personId = newPerson.id;
  }

  // Create event
  if (data.date) {
    await db.insert(events).values({
      userId,
      personId: personId ?? null,
      type: 'custom',
      title: data.eventName ?? suggestion.emailSubject,
      date: data.date,
      recurring: false,
      source: 'gmail_import',
      description: data.location ? `Location: ${data.location}` : undefined,
    });
  }

  await db.update(importSuggestions)
    .set({ status: 'approved' })
    .where(eq(importSuggestions.id, importId));

  revalidatePath('/imports');
  revalidatePath('/');
}

export async function batchApproveImports(importIds: string[]) {
  for (const id of importIds) {
    await approveImport(id);
  }
}

export async function dismissImport(importId: string) {
  const { userId } = await auth();
  if (!userId) throw new Error('Unauthorized');

  await db.update(importSuggestions)
    .set({ status: 'dismissed' })
    .where(and(eq(importSuggestions.id, importId), eq(importSuggestions.userId, userId)));

}

import { db } from '@/lib/db';
import { users, persons } from '@/lib/db/schema';
import { eq, count } from 'drizzle-orm';

export async function getUserProfile(userId: string) {
  return db.query.users.findFirst({ where: eq(users.id, userId) });
}

export async function isNewUser(userId: string): Promise<boolean> {
  const [result] = await db.select({ count: count() }).from(persons).where(eq(persons.ownerUserId, userId));

}

- [ ] **Step 4: Create push subscribe API route**

`app/api/push/subscribe/route.ts`:
```ts
import { auth } from '@clerk/nextjs/server';
import { db } from '@/lib/db';
import { pushSubscriptions } from '@/lib/db/schema';

export async function POST(req: Request) {
  const { userId } = await auth();
  if (!userId) return new Response('Unauthorized', { status: 401 });

  const { endpoint, keys } = await req.json();

  await db.insert(pushSubscriptions).values({
    userId,
    endpoint,
    p256dh: keys.p256dh,
    auth: keys.auth,
  }).onConflictDoNothing();

  return Response.json({ ok: true });
}
```
